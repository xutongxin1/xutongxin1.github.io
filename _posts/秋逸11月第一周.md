二、高精度阶乘计算

#include <stdio.h>

int a[100000]; 
int main()
{
int n; 
int jinwei; 
int weishu = 1; 
int temp, i, j; 
scanf("%d", &n); 
a[0] = 1;
for (i = 1; i <= n; i++) 
{

for (j = 1, jinwei = 0; j <= weishu; j++)
{
temp = a[j - 1] * i+jinwei ; 
a[j - 1] = temp % 10; 
jinwei = temp / 10; 
}
while (jinwei) 
{
a[++weishu - 1] = jinwei % 10; 
jinwei /= 10; 
}
}
printf("结果是:\n"); 
for (i = weishu; i >= 1; i--)
{
printf("%d", a[i - 1]);
}
}
